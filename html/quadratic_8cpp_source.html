107solving_state_t solve_quadratic(quadratic_equation_t *equation) {
108 assert(equation != NULL);
109
110 if(!isfinite(equation->a)) return INVALID_COEFFICIENTS;
111 if(!isfinite(equation->b)) return INVALID_COEFFICIENTS;
112 if(!isfinite(equation->c)) return INVALID_COEFFICIENTS;
113
114 //equation is linear if a == 0
115 if(is_zero(equation->a))
116 return solve_linear(equation);
117
118
119 double discriminant = equation->b * equation->b - 4 * equation->a * equation->c;
120
121 switch(compare_with_zero(discriminant)) {
122 case BIGGER: {
123 double discriminant_root = sqrt(discriminant);
124 equation->number = TWO_ROOTS;
125 equation->x1 = (-equation->b - discriminant_root) / (2 * equation->a);
126 equation->x2 = (-equation->b + discriminant_root) / (2 * equation->a);
127 return SOLVING_SUCCESS;
128 }
129 case EQUALS: {
130 equation->number = ONE_ROOT;
131 equation->x1 = equation->x2 = (-equation->b) / (2 * equation->a);
132 return SOLVING_SUCCESS;
133 }
134 case LESS: {
135 equation->number = NO_ROOTS;
136 return SOLVING_SUCCESS;
137 }
138 default: {
139 equation->number = NOT_SOLVED;
140 return SOLVING_ERROR;
141 }
142 }
143}

154void print_quadratic_result(const quadratic_equation_t *equation) {
155 assert(equation != NULL);
156 printf("Equation %lgx^2 + %lgx + %lg:\n",
157 equation->a, equation->b, equation->c);
158 switch(equation->number) {
159 case NOT_SOLVED: {
160 printf("Not solved yet, try to run solve_equation(...)\n");
161 return ;
162 }
163 case NO_ROOTS: {
164 printf("Does not have real roots\n");
165 return ;
166 }
167 case ONE_ROOT: {
168 printf("Has one root: x = %lg\n", equation->x1);
169 return ;
170 }
171 case TWO_ROOTS: {
172 printf("Has two roots: x1 = %lg, x2 = %lg\n", equation->x1, equation->x2);
173 return ;
174 }
175 case INF_ROOTS: {
176 printf("Has infinitely many roots\n");
177 return ;
178 }
179 default: {
180 printf("Something went wrong while trying to print out result\n");
181 return ;
182 }
183 }
184}
